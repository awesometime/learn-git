### 递归版本

```python3
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    # 中序遍历
    def Convert(self, pRootOfTree):
        # write code here
        if not pRootOfTree:
            return None
        
        if not pRootOfTree.left and not pRootOfTree.right:
            return pRootOfTree
        
        # 不用返回
        self.Convert(pRootOfTree.left)    # 将指针指好 返回值并没啥用 只要知道头结点就行
        left = pRootOfTree.left
        
        # 与下边的方法不同之处是:
        # 这个还是在操作树                 找到左子树的最右          不需要返回
        # 而下边是在操作Convert返回的链表   从头结点找到链表尾节点    需要返回
        
        if left:
            while left.right:
                left = left.right
                
            pRootOfTree.left = left    # 根的左边指向 left子树的最right
            left.right = pRootOfTree   # left子树的最right 指向根
            
        self.Convert(pRootOfTree.right)
        right = pRootOfTree.right
        
        if right:
            while right.left:
                right = right.left
            
            pRootOfTree.right = right
            right.left = pRootOfTree
        
        # 找到最左边 设置头结点
        while pRootOfTree.left:
            pRootOfTree = pRootOfTree.left
            
        return pRootOfTree
```

```python3
# https://www.nowcoder.com/questionTerminal/947f6eb80d944a84850b0538bf0ec3a5

class Solution:
    def Convert(self, root):
        if not root:
            return None
        if not root.left and not root.right:
            return root
         
        # 将左子树构建成双链表，返回链表头
        left = self.Convert(root.left)
        p = left
        
        # 与上边的方法不同之处是:
        # 上边是在操作树                   找到左子树的最右         不需要返回
        # 而这个是在操作Convert返回的链表   从头结点找到链表尾节点    需要返回
        
        # 定位至左子树的最右的一个结点
        while left and p.right:
            p = p.right
         
        # 如果左子树不为空，将当前root加到左子树链表
        if left:
            p.right = root             # a右是b  b左是a 应该可以换顺序
            root.left = p
         
        # 将右子树构造成双链表，返回链表头
        right = self.Convert(root.right)
        # 如果右子树不为空，将该链表追加到root结点之后
        if right:
            right.left = root          # a右是b  b左是a 应该可以换顺序
            root.right = right
             
        return left if left else root
```
### 非递归版本
```python3
# 链接：https://www.nowcoder.com/questionTerminal/947f6eb80d944a84850b0538bf0ec3a5

class Solution:
    def Convert(self, pRootOfTree):
        if not pRootOfTree:
            return None
         
        p = pRootOfTree
         
        stack = []
        resStack = []
         
        while p or stack:
            if p:
                stack.append(p)
                p = p.left
            else:
                node = stack.pop()
                resStack.append(node)
                p = node.right
             
        resP = resStack[0]
        while resStack:
            top = resStack.pop(0)
            if resStack:
                top.right = resStack[0]
                resStack[0].left = top
         
        return resP
```
### 先中序遍历，将所有的节点保存到一个列表中
```python3
# https://www.nowcoder.com/questionTerminal/947f6eb80d944a84850b0538bf0ec3a5

# 先中序遍历，将所有的节点保存到一个列表中。对这个list进行遍历，
# 每个节点的right设为下一个节点，下一个节点的left设为上一个节点。

class Solution:
    def Convert(self, pRootOfTree):
        # write code here
        if not pRootOfTree:return
        self.arr = []
        self.midTraversal(pRootOfTree)
        for i,v in enumerate(self.arr[:-1]):
            v.right = self.arr[i + 1]
            self.arr[i + 1].left = v
        return self.arr[0]
 
    def midTraversal(self, root):
        if not root: return
        self.midTraversal(root.left)
        self.arr.append(root)
        self.midTraversal(root.right)
```
### 非递归、O(1)空间复杂度的解法。该方法通过Morris遍历实现（搜索Morris Traversal），将二叉树重构为所有结点只有右子树的一条链
```java
# https://www.nowcoder.com/questionTerminal/947f6eb80d944a84850b0538bf0ec3a5

public class Solution {
    public TreeNode Convert(TreeNode pRootOfTree) {
        TreeNode p = pRootOfTree, pre = null, res = null;
        while (p != null) {
            while (p.left != null) {
                TreeNode q = p.left;
                while (q.right != null) {
                    q = q.right;
                }
                q.right = p;
                TreeNode tmp = p.left;
                p.left = null;
                p = tmp;
            }
            p.left = pre;
            if (pre == null) {
                res = p;
            } else {
                pre.right = p;
            }
            pre = p;
            p = p.right;
        }
        return res;
    }
}
```
