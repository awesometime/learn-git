import collections
# Runtime: 48 ms, faster than 84.00%

class Solution:
    def containsNearbyDuplicate(self, nums, k):
        if len(nums) <= 1: return False
        if k <= 0: return False
        lookup = []

        has = False
        for i in range(len(nums)):
            if nums[i] in lookup:
                return True
            if nums[i] not in lookup:
                lookup.append(nums[i])
                if len(lookup) <= k:    # 维护一个k的滑窗
                    continue
                else:
                    lookup.pop(0)
        return has
