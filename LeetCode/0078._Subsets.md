```python3
class Solution:
    def subsets(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        self.res = []
        def dfs(nums,temp,i):
            # print('---------------')
            self.res.append(temp[:])     # 【temp,  temp[:]区别】 temp[:]是深拷贝
            # print(' res|   '+str(res))
            for i in range(i,len(nums)):
                temp.append(nums[i])
                # print("temp|   " + str(temp))
                
                dfs(nums,temp,i+1)
                
                temp.pop()               # 需要pop()
                # print("temp|   " + str(temp))
                
        dfs(nums,[],0)
        return self.res
```

def subsets(nums):
    """
    :type nums: List[int]

            dfs(nums, temp+[nums[i]], i + 1)

    return res

### 4 Bit Manipulation

```python3 
# 总共有2^len(nums) 中情形
def subsets2(self, nums):
    res = []
    nums.sort()
    for i in xrange(1<<len(nums)):     # 1左移len(nums)位,等于2^len(nums)
        tmp = []
        for j in xrange(len(nums)):   # j= 0, 1, 2...n
            if i & 1 << j:  # if i >> j & 1:
                tmp.append(nums[j])
        res.append(tmp)
    return res
```
