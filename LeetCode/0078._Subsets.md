def subsets1(nums):
    res = []
    dfs(sorted(nums), 0, [], res)

### 2 Bit Manipulation

```python3   
def subsets2(self, nums):
    res = []
    nums.sort()
    for i in xrange(1<<len(nums)):
        tmp = []
        for j in xrange(len(nums)):
            if i & 1 << j:  # if i >> j & 1:
                tmp.append(nums[j])
        res.append(tmp)
    return res
```
