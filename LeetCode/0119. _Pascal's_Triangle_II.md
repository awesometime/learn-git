def getRow(rowIndex: 'int') -> 'List[int]':
    rpta = [1 for i in range(rowIndex + 1)]

    for i in range(2, rowIndex + 1):
        aux = rpta[0]
        for j in range(1, i):
            # rpta[j], aux = rpta[j] + aux, rpta[j]

            tmp = rpta[j]
            rpta[j] = rpta[j] + aux
            aux = tmp
    return rpta
