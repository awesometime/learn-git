class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """

        def preorder(root):
            res = []
            if not root:
                return res
            res.append(root.val)
            if root.left:
                res += preorder(root.left)
            if root.right:
                res += preorder(root.right)
            return res

        if not root:
            return
        node_list = preorder(root)
        cur_node = root
        for i in range(len(node_list)-1):
            node = TreeNode(node_list[i+1])
            cur_node.left = None
            cur_node.right = node
            cur_node = node
        cur_node.left = None
        cur_node.right = None

class Solution:
    def flatten(self, root: 'TreeNode') -> 'None':
        """
        Do not return anything, modify root in-place instead.
        """
        last=TreeNode(-1)
        qstack=[root]
        while qstack:
            node=qstack.pop()
            last.right=node
            last.left=None
            if node and node.right:    # 先append, pop时候后出来
                qstack.append(node.right)
            if node and node.left:     # 后入先出
                qstack.append(node.left)
            last=node

class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        preorder  root--> L --> R  右子树一定在左子树的后边,所以:
        1 将左子节点的最后一个右子节点的右节点 = 右子树
        2 当前根节点的右子节点 = 当前根节点的左子节点
        3 当前根节点的左子节点 = None

        """
        cur = root
        while cur:
            if cur.left:
                p = cur.left
                while p.right:
                    p = p.right
                p.right  = cur.right   # 1
                # cur.right = p  这样不对 因为while  p.right:中p会一直下沉 并不一定还是指cur.left
                cur.right = cur.left   # 2
                cur.left  = None       # 3
            cur = cur.right
