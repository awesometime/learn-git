# property setter
```python3
class Node():
    def __init__(self, data=None):
        self._data = data
        self._left, self._right = None, None

    def __str__(self):
        return 'Node:<data:%s>, <left:%s>, <right:%s>' % (
            str(self._data), str(self._left), str(self._right))

    @property
    def data(self):
        return self._data

    @data.setter
    def data(self, value):
        self._data = value

    @property
    def left(self):
        return self._left

    @left.setter
    def left(self, value):
        self._left = value

    @property
    def right(self):
        return self._right

    @right.setter
    def right(self, value):
        self._right = value
```
