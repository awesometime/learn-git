装饰器
```py3
import functools


def log(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print('call %s():' % func.__name__)
        print('args = {}'.format(*args))
        return func(*args, **kwargs)

    return wrapper

@log
def test(p):
    print(test.__name__ + " param: " + p)
    
p = "I'm a param"
test(p)


########### 等价 ################
def test(p):
    print(test.__name__ + " param: " + p)

p = "I'm a param"
wrapper = log(test)
wrapper(p)

# p = "I'm a param"
# log(test)(p) = wrapper(p)  wrapper实际上可以理解成具体的 test
```
