```go
    "fmt"
)

func main() {
    a:=0
    b:=0
    for {
        n, _ := fmt.Scan(&a,&b)
        if n == 0 {
            break
        } else {
            fmt.Printf("%d\n",a+b)
        }
    }
}
```

```go
package main

import (
    "fmt"
)
func main() {
    n:=0
    ans:=0

    fmt.Scan(&n)
    for i := 0; i < n; i++ {
        for j := 0; j < n; j++ {
            x:=0
            fmt.Scan(&x)
            ans = ans + x
        }
    }
    fmt.Printf("%d\n",ans)
}
```

```py3
import sys
if __name__ == "__main__":
    # 读取第一行的n
    n = int(sys.stdin.readline().strip())
    ans = 0
    for i in range(n):
        # 读取每一行
        line = sys.stdin.readline().strip()
        # 把每一行的数字分隔后转化成int列表
        values = list(map(int, line.split()))
        for v in values:
            ans += v
    print(ans)
```

```cpp
// 本题为考试多行输入输出规范示例，无需提交，不计分。
#include <iostream>
#include <cstdio>

using namespace std;

int main(){
    //freopen("1.in","r",stdin);
    int n,ans = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            int x; scanf("%d",&x);
            ans += x;
        }
    }
    cout << ans << endl;
    return 0;
}
```
