### 递归
```python3
def Merge(self, pHead1, pHead2):
    if not pHead1:
        return pHead2
    if not pHead2:
        return  pHead1
    if pHead1.val <= pHead2.val:
        pHead1.next = self.Merge(pHead1.next, pHead2)
        return pHead1
    else:
        pHead2.next = self.Merge(pHead1, pHead2.next)
        return pHead2

class Solution:
    # 返回合并后列表
    def Merge(self, pHead1, pHead2):
        # write code here
        if pHead1 == None:
            return pHead2
        if pHead2 == None:
            return pHead1
        
        pMergeHead = None
        if pHead1.val < pHead2.val:
            pMergeHead = pHead1
            pMergeHead.next = self.Merge(pHead1.next,pHead2)
        else:
            pMergeHead = pHead2
            pMergeHead.next = self.Merge(pHead1,pHead2.next)
        return pMergeHead
```

### 非递归

```python3
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    # 返回合并后列表
    def Merge(self, pHead1, pHead2):
        # write code here
        dummy = ListNode(0)
        pHead = dummy
         
        while pHead1 and pHead2:
            if pHead1.val >= pHead2.val:
                dummy.next = pHead2
                pHead2 = pHead2.next
            else:
                dummy.next = pHead1
                pHead1 = pHead1.next
                 
            dummy = dummy.next
        
        # 如果一个链表遍历完了，直接将另一个链表剩余部分链接到 cur 后面
        # 注意：当由于其中一链表listNode1或者listNode2为null，导致跳出while循环时，
        # 此时，还需要将另一不为null的链表的后续部分赋给合并链表。 
        if pHead1:
            dummy.next = pHead1
        elif pHead2:
            dummy.next = pHead2
         
        return pHead.next
```
