def isSymmetric(self, root):
    def sym_tree(L,R):                        # è¾…åŠ©å‡½æ•°
        if L and R:                           # å·¦        å³                   å³       å·¦
            return L.val == R.val and sym_tree(L.left, R.right) and sym_tree(L.right, R.left)
        else:
            return L == R
    return sym_tree(root, root)
