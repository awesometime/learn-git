def getRow(rowIndex: 'int') -> 'List[int]':
    rpta = [1 for i in range(rowIndex + 1)]

    for i in range(2, rowIndex + 1):
        aux = rpta[0]

    return rpta

def getRow(self, rowIndex):
    """
    :type rowIndex: int
    :rtype: List[int]
    """
    res = [1]
    for i in range(1, rowIndex + 1):
        res = [1] + [res[i] + res[i + 1] for i in range(len(res) - 1)] + [1]
    return res
